分治思想处理最大子序列

核心思想就是二分法无限不断持续细分，用递归来实现，把每一个细分板块都看作独立的数列处理：

int Maxsequence(int array[],int left,int right){
    int center,i;
    int maxleftsum,maxrightsum;   //左半边最大子序列&右半边最大子序列
    int leftsum,mleftsum,rightsum,mrightsum;   //两边分别处理  
    if (left==right){    //基础情况：只有一个元素
        return array[left];
    }
    //先判断完成基础情况再进行递归
    center=(left+right)/2;     
    maxleftsum=Maxsequence(array,left,center);    //递归左半边
    maxrightsum=Maxsequence(array,center+1,right);  //递归右半边
    leftsum=0;mleftsum=0;  //别忘了初始化
    for (i=left;i<=center;i++){
        leftsum+=array[i];
        if (leftsum>mleftsum){
            mleftsum=leftsum;
        }
    }
    rightsum=0;mrightsum=0;
    for (i=center+1;i<=right;i++){
        rightsum+=array[i];
        if (rightsum>mrightsum){
            mrightsum=rightsum;
        }
    }
    return max3(maxleftsum,maxrightsum,mrightsum+mleftsum);   //左半边最大子序列、右半边最大子序列和中间最大子序列（发现规律得来）
    }

动态规划处理最大子序列

从开头开始，遇到负数就ban掉，和是正数就继续遍历，如果数列第一个元素是负数则重来（和为0）

int online-maxsequence(int array[],int n){
    int sum=0,maxsum=0;
    int i;
    for (i=0;i<n;i++){
        sum+=array[i];  
        if (sum>maxsum)
            maxsum=sum;
        else if (sum<0)   //换个角度和为负数则归零
        sum=0;
    }
    return maxsum;
}

链表实现多项式相加

创建链表、实现相加操作、打印结果链表

//链表结构定义
struct polynomial{
    int xishu;
    int zhishu;
    struct polynomial* next;
}
typedef struct polynomial* poly;

//构造链表新节点
poly createnode(poly head,int a,int b){
    poly newnode=new polynomial;
    newnode->xishu=a;
    newnode->zhishu=b;
    newnode->next=NULL;
    poly ptr;
    if (head=NULL)
    head=newnode;
    else {
        ptr=head;
        while (ptr->next!=NULL){
            ptr=ptr->next;
        }
        ptr->next=newnode;
    }
    return head;
}

//多项式相加操作
poly addition(poly nums1,poly nums2){
    poly nums=NULL;  /*/链表初始化，切记勿忘
    while (nums1!=NULL&&nums2!=NULL){  //多项式公用部分
        if (nums1->zhishu==nums2->zhishu){
            nums=createnode(nums,nums1->xishu+nums2->xishu,nums1->zhishu);    /*/nums=createnode...而不是直接createnode...
            nums1=nums1->next;  /*/遍历基本条件勿忘
            nums2=nums2->next;  /*/
        }
        else if (nums1->zhishu<nums2->zhishu){  
            nums=createnode(nums,nums1->xishu,nums1->zhishu);
            nums1=nums1->next;  /*/
        }
        else {
            nums=createnode(nums,nums2->xishu,nums2->zhishu);
            nums2=nums2->next;   /*/
        }
    }
    while (nums1!=NULL){  //公用结束后单独部分
        nums=createnode(nums,nums1->xishu,nums1->zhishu);
        nums1=nums1->next;
    }
    while (nums2!=NULL){  //共用结束后单独部分
        nums=createnode(nums,nums2->xishu,nums2->zhishu);
        nums2=nums2->next;
    }
    return nums;
}

//打印，根据系数是否为0进行打印，不为0则判断指数是否为0，打印元素，遍历，每次循环最后判断系数部位0则打印加号
void print(poly sum){
    while (sum!=NULL){  
        if (sum->xishu!=0){
        cout<<sum->xishu;
        if (sum->zhishu!=0)
        cout<<"x^"<<sum->zhishu;
    }
    sum=sum->next;
    if (sum!=NULL&&sum->xishu!=0)
    cout<<"+";
    }
    cout<<endl;
}

反转单链表

把指针反转就可

//反转操作：需要三个指针，prev和curr进行反转操作，next储存下一次循环的curr
typedef struct linklist* ll;
ll (ll nums){
ll prev,curr,next;
curr=nums;
while (curr->next!=NULL){
    next=curr->next;
    curr->next=prev;
    prev=curr;
    curr=next;
}
return prev;
}

约瑟夫环

构建循环链表需要头尾和当下元素，再来一个暂时指针用于链表的释放；

//构建结构体
struct people
    int number;
    struct people* next;
}*head,*tail,*curr,*temp;
int main(){
    int n,m;   //n表示人数，m表示循环淘汰的数字
    cin>>n>>m;
    int i,j;
    head=new people;  //头指针不存放任何信息
    head->next=NULL;
    tail=head;       //尾插法
    for (i=1;i<=n;i++){
        curr=new people;
        curr->number=i;
        curr->next=NULL;
        tail->next=curr;
        tail=curr;  //更新尾指针
    }
    curr=head->next;  //连接头指针和数据链表
    tail->next=curr;  //尾指针连接数据链表第一个指针，构成循环链表
    for (i=0;i<n;i++){
        for (j=1;j<m-1;j++){   //循环到去除元素前一位元素（方便存储和释放）
            curr=curr->next;
        }
        cout<<curr->next->number<<" ";
        temp=curr->next;   //暂时指针记录去除元素，删除后释放空间
        curr->next=temp->next;
        curr=curr->next;
        free(temp);
    }
    return 0;
}

链表创建双端队列

队列的特点是先进先出，双端队列即一个链表可以从前端插入删除，也可以从后端插入删除。看着多，实则写起来不难。

struct queue{
    int number;
    struct queue* next;
    struct queue* prev;
}
typedef struct queue* sq;
sq fronts=NULL;
sq backs=NULL;
//从前端插入元素
void push(sq q,int a){
    sq newnode=new queue;
    newnode->number=a;
    newnode->next=fronts;
    newnode->prev=NULL;
    if (backs==NULL){
        fronts=newnode;
        backs=newnode;
    }
    else {
    fronts->prev=newnode;
    fronts=newnode;
    }
}

//从前端删除元素,并返回被删除的元素
int pop(sq q){
    sq temp=fronts;
    int result=temp->number;
    if (fronts!=backs){
        fronts=fronts->next;
        fronts->prev=NULL;
    }
    else if (backs==fronts){
        backs=NULL;
        fronts=NULL;
    }
    else {
        return -1;
    }
    delete temp;
    return result;
}

//从后端插入元素
void inject(sq q,int a){
    sq newnode=new queue;
    newnode->number=a;
    newnode->next=NULL;
    newnode->prev=backs;
    if (fronts==NULL){
        backs=newnode;
        fronts=newnode;
    }
    else {
        backs->next=newnode;
        backs=newnode;
    }
}

//从后端删除元素，并返回被删除的元素
int eject(sq q){
    sq temp=backs;
    int result=temp->number;
    if (backs==fronts){
        backs=NULL;
        fronts=NULL;
    }
    else if (backs!=fronts){
        backs=temp->prev;
        backs->next=NULL;
    }
    else {
        return -1;
    }
    delete temp;
    return result;
}

数组创建双栈

栈的特点是先进后出；该程序要求判断前后端队列是否是空的、前端插入删除元素、后端插入删除元素，使用index指明是从前端还是从后端进行处理

int maxnumber=100;
int s[maxnumber]={0};
int fronts=0;
int backs=maxnumber;

//分别判断前后端队列是否是空的
bool isempty(int index){
    if (index==1&&fronts<1)
    return true;
    if (index==2&&backs>maxnumber-1)
    return true;
    return false;
}

//从前后端插入元素
void push(int index,int data){
    if (fronts>backs){
        return;
    }
    else if (index==1){
        fronts++;
        s[fronts]=data;
    }
    else if (index==2){
        backs--;
        s[backs]=data;
    }
}

//从前后端删除元素
int pop(int index){
    int result;
    if (index==1){
    if (fronts<0){
        return -1;
    }
    else {
        result=s[fronts];
        fronts--;
    }
    }
    else {
        if (backs>maxnumber)
        return -1;
        else {
            result=s[backs];
            backs++;
        }
    }
    return result;
}
